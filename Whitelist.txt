# Формат: инжектор:device_id:user_id
# * означает любое значение

# Разрешить все для Synapse X с определенным device_id
Synapse X:DEV_123456:*

# Разрешить пользователю 123456789 с любым инжектором и device_id
*:*:123456789

# Разрешить конкретную комбинацию
Xeno:DEV_789012:987654321

# Разрешить Vulcan для всех устройств и пользователей
Vulcan:*:*

# Разрешить Oxygen U для определенного устройства
Oxygen U:DEV_345678:*

# Просто по ID пользователя (устаревший формат, но поддерживается)
555555555
666666666
